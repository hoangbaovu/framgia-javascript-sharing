JavaScript Variable

var, let and const .

var is scoped to the nearest function block and let is scoped to the nearest enclosing ({}) block.

var and let scoping


							const varTest = () => {
								var x = 1;
								if (true) {
									var x = 2;
									console.log(x);
								}
								console.log(x);
							}

							varTest();
						

								const letTest = () => {
									let x = 1;
									if (true) {
										let x = 2;
										console.log(x);
									}
									console.log(x);
								}

								letTest();

Hoisting In JavaScript

All variable declaration are hoisted to the top of the function.

Noted: Only variable declarations are hoisted to the top, not variable assignment (when the variable is assigned a value) 


							var a = 'Hello Holy';
						


							// javascript hoisting
							var a;
							a = 'Hello Holy';

Variable Hoisting


							// real code
							funcB();
							console.log(a);

							var a = 'Hello World';
							function funcB() {
								console.log('Called b!');
							}
						

							// javascript hoisting
							function funcB() {
								console.log('Called b!');
							}
							funcB();
							var a;
							console.log(a);
							a = 'Hello World';

JavaScript Objects

An Object is an unordered list of primitive data types that is stored as a series of name-value pairs.


							const obj = {
								name: 'Holy',
								age: 26,
								showName: () => console.log(this.name),
							};

Object property name

Property name can be a string or number.


							var ageGroup = {
								30: "Children",
								100:"Very Old",
							};
							console.log(ageGroup.30) // This will throw an error​

							​// if the property name is a number,
							// it has to be accessed with the bracket notation
							console.log(ageGroup["30"]); // Children​

Reference Data Type and Primitive Data Types

Reference data type’s value is stored as a reference.

Primitive data type's value is stored directly on the variable.

Primitive data types


							var person = 'Holy';
							var anotherPerson = person;
							person = 'Tony';

							console.log(person); // Tony
							console.log(anotherPerson); // Holy

Reference data types


							var person = { name: 'Holy' };
							var anotherPerson = person;
							person.name = 'Tony';

							console.log(person); // Tony
							console.log(anotherPerson); // Tony

Serialize and Deserialize Objects

Using JSON.stringify to Serialize (convert object to a string) Using JSON.parse to Deserialize (convert string to object) 


							var userInfo = {
								username: 'tinhyeucuatoi',
								email: 'tinhyeucuatoi@gmail.com',
								password: 'xxxxxx'
							};

							var serialize = JSON.stringify(userInfo);
							// {"username":"tinhyeucuatoi","email":"tinhyeucuatoi@gmail.com","password":"xxxxxx"}
							var deserialize = JSON.parse(serialize);
							// {username: "tinhyeucuatoi", email: "tinhyeucuatoi@gmail.com", password: "xxxxxx"}

JavaScript Prototype

Every JavaScript function has a prototype property and you attach properties and methods on this prototype when you want to implement inheritance

JavaScript Prototype


							function Animal(name) {
								this.name = name;
							}

							Animal.prototype.getName = function() {
								console.log(`Hello, My name is ${this.name}`);
							}

							var cat = new Animal('Kitty');
							cat.getName(); // Hello, My name is Kitty

							var duck = new Animal('Delly');
							duck.getName(); // Hello, My name is Delly

JavaScript Prototype

In JavaScript, you add methods and properties on the prototype property when you want instances of an object to inherit those methods and properties

OOP in JavaScript

  • Inheritance
  • Abstraction
  • Encapsulation

OOP Pattern


							function Employee(name, email) {
								this.name = name;
								this.email = email;
								this.workingDay = [];
							}

							Employee.prototype = {
								constructor: Employee,
								changeEmail: function(newEmail) {
									this.email = newEmail;
								},
								addWorkingDay: function(newValue) {
									this.workingDay.push(newValue);
									return this.workingDay;
								},
								getEmail: function() {
									return this.email;
								}
							}

							var tony = new Employee('Tony', 'tinhyeucuatoi@gmail.com');
							console.log(tony.__proto__); // {constructor: ƒ, changeEmail: ƒ, addWorkingDay: ƒ, getEmail: ƒ}
							tony.changeEmail('tony1102@gmail.com');
							tony.getEmail(); // tony1102@gmail.com

							var holy = new Employee('Holy', 'holy@gmailcom');
							holy.getEmail(); // holy@gmail.com

JavaScript Closure

A closure is an inner function that has access to the outer (enclosing) function’s variables.

A closure happens when you add a function inside another function.

JavaScript Closure


								function user(firstName, lastName) {
									var outerVariable = 'Your name is ';

									function getFullName() {
										return outerVariable + firstName + ' ' + lastName;
									}

									return getFullName;
								}

								var getFullName = user('Tony', 'Hood');
								console.log(getFullName());

JavaScript Closure


						function funcReturnObject() {
							var id = 10;

							return {
								getId: () => id,
								setId: (newId) => {
									id = newId;
								}
							};
						}

						var resultObj = funcReturnObject();
						resultObj.getId(); // 10

						resultObj.setId(999);
						resultObj.getId(); // 999
